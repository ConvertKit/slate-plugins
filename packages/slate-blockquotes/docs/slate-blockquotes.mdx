A plugin for blockquotes in SlateJS.

## Install

```bash
yarn add @convertkit/slate-blockquotes
```

## Usage

```jsx
import { Editor } from "slate-react";
import Blockquotes from "@convertkit/slate-blockquotes"

const plugins = [Blockquotes({
  blocks: {
    blockquote: "blockquote",
    blockquote_line: "blockquote-line",
    default: "paragraph"
  },
  classNames: {
    blockquote: "blockquote",
    blockquote_line: "blockquote-line",
  }
})]

const MyEditor = (props) => (
  <Editor
    ...
    plugins={plugins}
  />
)
```
